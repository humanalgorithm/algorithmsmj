class InsertionSort(object):
    def insertion_sort(self, dataset):
        for sorted_index in range(1, len(dataset)):
            j = sorted_index
            while j >= 1 and dataset[j] < dataset[j - 1]:
                temp = dataset[j - 1]
                dataset[j - 1] = dataset[j]
                dataset[j] = temp
                j = j - 1
        return dataset
